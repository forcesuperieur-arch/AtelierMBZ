// Calculateur de Devis - Atelier Moto Pro // Compatible avec core.js const API_URL_DEVIS = window.API_URL || window.location.origin; let clients = []; let vehicules = []; let lignes = []; // Helper pour les requêtes API async function apiRequestDevis(url, options = {}) { const token = localStorage.getItem('token'); const defaultOptions = { headers: { 'Authorization': `Bearer ${token || ''}`, 'Content-Type': 'application/json' } }; const response = await fetch(`${API_URL_DEVIS}${url}`, { ...defaultOptions, ...options, headers: { ...defaultOptions.headers, ...options.headers } }); if (response.status === 401) { localStorage.removeItem('token'); alert('Session expirée. Veuillez vous reconnecter.'); window.location.href = '/dashboard.html'; throw new Error('Session expirée'); } if (!response.ok) { throw new Error(`Erreur ${response.status}`); } return response.json(); } // Chargement initial document.addEventListener('DOMContentLoaded', () => { chargerClients(); document.getElementById('clientSelect').addEventListener('change', onClientChange); }); // Charger la liste des clients async function chargerClients() { try { clients = await apiRequestDevis('/api/clients'); const select = document.getElementById('clientSelect'); select.innerHTML = ''; clients.forEach(c => { select.innerHTML += ``; }); } catch (e) { console.error('Erreur chargement clients:', e); } } // Quand un client est sélectionné async function onClientChange(e) { const clientId = e.target.value; const selectVehicule = document.getElementById('vehiculeSelect'); selectVehicule.innerHTML = ''; if (!clientId) return; try { vehicules = await apiRequestDevis(`/api/clients/${clientId}/vehicules`); vehicules.forEach(v => { selectVehicule.innerHTML += ``; }); } catch (e) { console.error('Erreur chargement véhicules:', e); } } // Ajouter une ligne (prestation ou pièce) function ajouterLigne(type) { const ligne = { id: Date.now(), type: type, designation: '', quantite: 1, prix_unitaire_ht: 0 }; lignes.push(ligne); renderLignes(); } // Supprimer une ligne function supprimerLigne(id) { lignes = lignes.filter(l => l.id !== id); renderLignes(); } // Mettre à jour une ligne function updateLigne(id, field, value) { const ligne = lignes.find(l => l.id === id); if (ligne) { ligne[field] = field === 'designation' ? value : parseFloat(value) || 0; } } // Afficher les lignes dans le tableau function renderLignes() { const tbody = document.getElementById('lignesBody'); tbody.innerHTML = ''; lignes.forEach(ligne => { const tr = document.createElement('tr'); tr.innerHTML = ` ${ligne.type === 'piece' ? '🔧 Pièce' : '👨‍🔧 Prestation'} `; tbody.appendChild(tr); }); } // Calculer le devis via API async function calculerDevis() { const clientId = document.getElementById('clientSelect').value; const vehiculeId = document.getElementById('vehiculeSelect').value; if (!clientId || !vehiculeId) { alert('Veuillez sélectionner un client et un véhicule'); return; } if (lignes.length === 0) { alert('Ajoutez au moins une ligne'); return; } // Vérifier que toutes les lignes ont une désignation for (const ligne of lignes) { if (!ligne.designation.trim()) { alert('Toutes les lignes doivent avoir une description'); return; } } const payload = { lignes: lignes.map(l => ({ type_ligne: l.type === 'piece' ? 'piece' : 'main_oeuvre_libre', description: l.designation, quantite: l.quantite, prix_unitaire_ht: l.prix_unitaire_ht })), remise_pourcentage: 0 }; try { const data = await apiRequestDevis('/api/devis/calculer', { method: 'POST', body: JSON.stringify(payload) }); afficherResultat(data); } catch (e) { alert('Erreur: ' + e.message); } } // Afficher le résultat du calcul function afficherResultat(data) { document.getElementById('resultat').style.display = 'block'; document.getElementById('totalHt').textContent = data.total_ht.toFixed(2) + ' €'; document.getElementById('tva').textContent = (data.total_ttc - data.total_ht).toFixed(2) + ' €'; document.getElementById('totalTtc').textContent = data.total_ttc.toFixed(2) + ' €'; // Afficher le détail const detailDiv = document.getElementById('detailLignes'); if (detailDiv) { detailDiv.innerHTML = `
Main d'œuvre HT: ${data.total_mo_ht.toFixed(2)} €
Pièces HT: ${data.total_pieces_ht.toFixed(2)} €
${data.remise_montant > 0 ? `Remise: -${data.remise_montant.toFixed(2)} €
` : ''} Total HT: ${data.total_ht_remise.toFixed(2)} €
Total TTC: ${data.total_ttc.toFixed(2)} €
`; } }
