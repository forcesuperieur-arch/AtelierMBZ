const API_URL = ''; let mecaniciens = []; let ponts = []; let absences = []; // Stockage des absences function getToken() { return localStorage.getItem('token') || ''; } async function apiGet(url) { const response = await fetch(`${API_URL}${url}`, { headers: { 'Authorization': `Bearer ${getToken()}` } }); if (response.status === 401) { localStorage.removeItem('token'); window.location.href = '/dashboard.html'; throw new Error('Session expirée'); } if (!response.ok) throw new Error(`Erreur ${response.status}`); return response.json(); } async function apiPost(url, data) { const response = await fetch(`${API_URL}${url}`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${getToken()}` }, body: JSON.stringify(data) }); if (response.status === 401) { localStorage.removeItem('token'); window.location.href = '/dashboard.html'; throw new Error('Session expirée'); } if (!response.ok) throw new Error(`Erreur ${response.status}`); return response.json(); } async function apiPut(url, data) { const response = await fetch(`${API_URL}${url}`, { method: 'PUT', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${getToken()}` }, body: JSON.stringify(data) }); if (response.status === 401) { localStorage.removeItem('token'); window.location.href = '/dashboard.html'; throw new Error('Session expirée'); } if (!response.ok) throw new Error(`Erreur ${response.status}`); return response.json(); } async function apiDelete(url) { const response = await fetch(`${API_URL}${url}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${getToken()}` } }); if (response.status === 401) { localStorage.removeItem('token'); window.location.href = '/dashboard.html'; throw new Error('Session expirée'); } if (!response.ok) throw new Error(`Erreur ${response.status}`); return response; } // Chargement async function chargerDonnees() { try { mecaniciens = await apiGet('/api/mecaniciens'); ponts = await apiGet('/api/ponts'); absences = await chargerAbsences(); // Charger les absences d'abord renderMeca(); renderPonts(); updateStats(); updateSelectMeca(); } catch (error) { console.error('Erreur:', error); } } function renderMeca() { const container = document.getElementById('liste-meca'); if (mecaniciens.length === 0) { container.innerHTML = '

Aucun mécanicien

'; return; } container.innerHTML = mecaniciens.map(m => `
${m.prenom.charAt(0)}${m.nom.charAt(0)}

${m.prenom} ${m.nom}

${m.specialites || 'Aucune spécialité'}

${m.actif ? 'Actif' : 'Inactif'}
`).join(''); } function renderPonts() { const container = document.getElementById('liste-ponts'); if (ponts.length === 0) { container.innerHTML = '

Aucun pont

'; return; } const today = new Date().toISOString().split('T')[0]; container.innerHTML = ponts.map(p => { const meca = mecaniciens.find(m => m.id === p.mecanicien_id); // Vérifier si le mécanicien est absent aujourd'hui const mecaAbsent = meca && absences.some(a => a.mecanicien_id === meca.id && a.date_debut <= today && a.date_fin >= today ); // Déterminer le statut réel du pont let statusClass, statusText, statusColor; if (!p.actif) { statusClass = 'pont-maintenance'; statusText = 'Maintenance'; statusColor = 'bg-yellow-100 text-yellow-800'; } else if (mecaAbsent) { statusClass = 'pont-maintenance'; // Orange comme maintenance statusText = 'Mécanicien absent'; statusColor = 'bg-orange-100 text-orange-800'; } else { statusClass = 'pont-libre'; statusText = 'Disponible'; statusColor = 'bg-green-100 text-green-800'; } return `

${p.nom}

${p.type_pont || 'Moto'} - ${p.capacite_kg || 500}kg

${meca ? `👤 ${meca.prenom} ${meca.nom}` : 'Non assigné'}

${statusText}
`; }).join(''); } function updateStats() { const today = new Date().toISOString().split('T')[0]; // Compter les mécaniciens actifs document.getElementById('count-meca').textContent = mecaniciens.filter(m => m.actif).length; // Compter les ponts disponibles (actifs ET mécanicien présent) const pontsDisponibles = ponts.filter(p => { if (!p.actif) return false; if (!p.mecanicien_id) return true; // Pont sans mécanicien = disponible // Vérifier si le mécanicien est absent const mecaAbsent = absences.some(a => a.mecanicien_id === p.mecanicien_id && a.date_debut <= today && a.date_fin >= today ); return !mecaAbsent; }); // Compter les ponts indisponibles (maintenance ou mécanicien absent) const pontsIndisponibles = ponts.filter(p => { if (!p.actif) return true; // Maintenance if (!p.mecanicien_id) return false; const mecaAbsent = absences.some(a => a.mecanicien_id === p.mecanicien_id && a.date_debut <= today && a.date_fin >= today ); return mecaAbsent; }); document.getElementById('count-ponts-libres').textContent = pontsDisponibles.length; document.getElementById('count-ponts-occ').textContent = pontsIndisponibles.length; // Compter les absences d'aujourd'hui const absencesToday = absences.filter(a => a.date_debut <= today && a.date_fin >= today); document.getElementById('count-abs').textContent = absencesToday.length; } function updateSelectMeca() { const select = document.getElementById('pont-meca'); select.innerHTML = '' + mecaniciens.filter(m => m.actif).map(m => `` ).join(''); } // Modals Mécanicien function openMecaModal(id = null) { document.getElementById('meca-id').value = id || ''; if (id) { const m = mecaniciens.find(x => x.id === id); document.getElementById('meca-nom').value = m.nom; document.getElementById('meca-prenom').value = m.prenom; document.getElementById('meca-spec').value = m.specialites || ''; document.getElementById('meca-couleur').value = m.couleur || '#3b82f6'; document.getElementById('meca-actif').value = m.actif ? '1' : '0'; } else { document.getElementById('meca-nom').value = ''; document.getElementById('meca-prenom').value = ''; document.getElementById('meca-spec').value = ''; document.getElementById('meca-couleur').value = '#3b82f6'; document.getElementById('meca-actif').value = '1'; } document.getElementById('modal-meca').classList.remove('hidden'); } function closeMecaModal() { document.getElementById('modal-meca').classList.add('hidden'); } async function saveMeca() { const id = document.getElementById('meca-id').value; const data = { nom: document.getElementById('meca-nom').value, prenom: document.getElementById('meca-prenom').value, specialites: document.getElementById('meca-spec').value, couleur: document.getElementById('meca-couleur').value, actif: parseInt(document.getElementById('meca-actif').value) }; try { if (id) { await apiPut(`/api/mecaniciens/${id}`, data); } else { await apiPost('/api/mecaniciens', data); } closeMecaModal(); chargerDonnees(); } catch (error) { alert('Erreur: ' + error.message); } } async function editMeca(id) { openMecaModal(id); } async function deleteMeca(id) { if (!confirm('Supprimer ce mécanicien ?')) return; try { await apiDelete(`/api/mecaniciens/${id}`); chargerDonnees(); } catch (error) { alert('Erreur: ' + error.message); } } // Modals Pont function openPontModal(id = null) { updateSelectMeca(); document.getElementById('pont-id').value = id || ''; if (id) { const p = ponts.find(x => x.id === id); document.getElementById('pont-nom').value = p.nom; document.getElementById('pont-type').value = p.type_pont || 'moto'; document.getElementById('pont-capacite').value = p.capacite_kg || 500; document.getElementById('pont-meca').value = p.mecanicien_id || ''; document.getElementById('pont-actif').value = p.actif ? '1' : '0'; } else { document.getElementById('pont-nom').value = ''; document.getElementById('pont-type').value = 'moto'; document.getElementById('pont-capacite').value = 500; document.getElementById('pont-meca').value = ''; document.getElementById('pont-actif').value = '1'; } document.getElementById('modal-pont').classList.remove('hidden'); } function closePontModal() { document.getElementById('modal-pont').classList.add('hidden'); } async function savePont() { const id = document.getElementById('pont-id').value; const data = { nom: document.getElementById('pont-nom').value, type_pont: document.getElementById('pont-type').value, capacite_kg: parseInt(document.getElementById('pont-capacite').value), mecanicien_id: document.getElementById('pont-meca').value || null, actif: parseInt(document.getElementById('pont-actif').value) }; try { if (id) { await apiPut(`/api/ponts/${id}`, data); } else { await apiPost('/api/ponts', data); } closePontModal(); chargerDonnees(); } catch (error) { alert('Erreur: ' + error.message); } } async function editPont(id) { openPontModal(id); } async function deletePont(id) { if (!confirm('Supprimer ce pont ?')) return; try { await apiDelete(`/api/ponts/${id}`); chargerDonnees(); } catch (error) { alert('Erreur: ' + error.message); } } // Gestion des absences function openAbsenceModal() { const select = document.getElementById('absence-meca'); select.innerHTML = '' + mecaniciens.filter(m => m.actif).map(m => `` ).join(''); const today = new Date().toISOString().split('T')[0]; document.getElementById('absence-debut').value = today; document.getElementById('absence-fin').value = today; document.getElementById('absence-notes').value = ''; document.getElementById('modal-absence').classList.remove('hidden'); } function closeAbsenceModal() { document.getElementById('modal-absence').classList.add('hidden'); } async function saveAbsence() { const data = { mecanicien_id: parseInt(document.getElementById('absence-meca').value), date_debut: document.getElementById('absence-debut').value, date_fin: document.getElementById('absence-fin').value, motif: document.getElementById('absence-motif').value, notes: document.getElementById('absence-notes').value }; if (!data.mecanicien_id) { alert('Veuillez sélectionner un mécanicien'); return; } try { await apiPost('/api/absences', data); closeAbsenceModal(); chargerAbsences(); } catch (error) { alert('Erreur: ' + error.message); } } async function chargerAbsences() { const date = document.getElementById('absence-date')?.value || new Date().toISOString().split('T')[0]; try { const absencesData = await apiGet(`/api/absences?date=${date}`); renderAbsences(absencesData); return absencesData; // Retourner pour utilisation dans d'autres fonctions } catch (error) { console.error('Erreur chargement absences:', error); return []; } } function renderAbsences(absences) { const container = document.getElementById('liste-absences'); if (!absences || absences.length === 0) { container.innerHTML = '

Aucune absence

✕

${meca ? meca.prenom + ' ' + escapeHtml(meca.nom) : 'Inconnu'}

${a.motif} - ${new Date(a.date_debut).toLocaleDateString('fr-FR')} au ${new Date(a.date_fin).toLocaleDateString('fr-FR')}

`; }).join(''); } async function deleteAbsence(id) { if (!confirm('Supprimer cette absence ?')) return; try { await apiDelete(`/api/absences/${id}`); chargerAbsences(); } catch (error) { alert('Erreur: ' + error.message); } } // Initialisation document.addEventListener('DOMContentLoaded', () => { if (!getToken()) { window.location.href = '/dashboard.html'; return; } chargerDonnees(); // Charger absences quand la date change document.getElementById('absence-date').addEventListener('change', chargerAbsences); });
