// Récupérer le token
function getToken() {
    const token = localStorage.getItem('token');
    if (!token) {
        window.location.href = '/';
        return null;
    }
    return token.trim().replace(/^[\"']|[\"']$/g, '');
}

// Appel API
async function apiGet(url) {
    const token = getToken();
    if (!token) return null;
    
    const response = await fetch(`${API_URL}${url}`, {
        headers: { 'Authorization': `Bearer ${token}` }
    });
    
    if (response.status === 401) {
        localStorage.removeItem('token');
        window.location.href = '/';
        throw new Error('Session expirée');
    }
    
    if (!response.ok) throw new Error(`Erreur ${response.status}`);
    return response.json();
}

async function apiPut(url, data) {
    const token = getToken();
    if (!token) return null;
    
    const response = await fetch(`${API_URL}${url}`, {
        method: 'PUT',
        headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${token}`
        },
        body: JSON.stringify(data)
    });
    
    if (response.status === 401) {
        localStorage.removeItem('token');
        window.location.href = '/';
        throw new Error('Session expirée');
    }
    
    if (!response.ok) throw new Error(`Erreur ${response.status}`);
    return response.json();
}

// Charger les RDV du technicien
async function loadTechRDV() {
    try {
        // Récupérer le planning du jour
        const today = new Date().toISOString().split('T')[0];
        const planning = await apiGet(`/api/planning?date=${today}`);
        
        if (!planning || !planning.rendez_vous) {
            showEmptyRDVList();
            return;
        }
        
        // Récupérer l'ID du mécanicien connecté (depuis le token ou une autre API)
        const mecanoId = await getCurrentMecanicienId();
        
        // Filtrer les RDV assignés à ce mécanicien
        let mesRDV = planning.rendez_vous;
        if (mecanoId) {
            mesRDV = mesRDV.filter(r => r.mecanicien_id === mecanoId);
        }
        
        // Trier par heure
        mesRDV.sort((a, b) => a.heure_rdv.localeCompare(b.heure_rdv));
        
        renderRDVList(mesRDV);
        updateStats(mesRDV);
        
    } catch (error) {
        console.error('Erreur chargement RDV:', error);
        document.getElementById('rdv-list').innerHTML = `
            <div class="text-center py-8 text-red-500">
                <i class="fas fa-exclamation-circle text-2xl mb-2"></i>
                <p>Erreur de chargement</p>
            </div>
        `;
    }
}

function renderRDVList(rdvs) {
    const container = document.getElementById('rdv-list');
    
    if (rdvs.length === 0) {
        showEmptyRDVList();
        return;
    }
    
    container.innerHTML = rdvs.map(rdv => {
        const statusClass = rdv.statut === 'en_cours' ? 'status-en_cours' : 
                           rdv.statut === 'termine' ? 'status-termine' : 'status-attente';
        const orSigne = rdv.kilometrage && rdv.etat_vehicule;
        
        return `
            <div class="rdv-card ${statusClass} bg-white rounded-lg p-4 ${currentRdvId === rdv.id ? 'active' : ''}" 
                 onclick="selectRDV(${rdv.id})" data-rdv-id="${rdv.id}">
                <div class="flex justify-between items-start">
                    <div>
                        <div class="font-bold text-lg">${rdv.heure_rdv}</div>
                        <div class="text-sm text-gray-600">${rdv.client.prenom} ${rdv.client.nom}</div>
                        <div class="text-xs text-gray-500">${rdv.vehicule.marque} ${rdv.vehicule.modele}</div>
                    </div>
                    <div class="text-right">
                        ${orSigne ? '<span class="text-green-600" title="OR signé"><i class="fas fa-clipboard-check"></i></span>' : ''}
                        <span class="text-xs px-2 py-1 rounded-full ${getStatusBadgeClass(rdv.statut)}">${getStatusLabel(rdv.statut)}</span>
                    </div>
                </div>
            </div>
        `;
    }).join('');
}

function getStatusBadgeClass(statut) {
    const classes = {
        'en_attente': 'bg-yellow-100 text-yellow-800',
        'confirme': 'bg-blue-100 text-blue-800',
        'en_cours': 'bg-orange-100 text-orange-800',
        'termine': 'bg-green-100 text-green-800',
        'facture': 'bg-green-100 text-green-800'
    };
    return classes[statut] || 'bg-gray-100 text-gray-800';
}

function getStatusLabel(statut) {
    const labels = {
        'en_attente': 'Attente',
        'confirme': 'Confirmé',
        'en_cours': 'En cours',
        'termine': 'Terminé',
        'facture': 'Facturé'
    };
    return labels[statut] || statut;
}

function updateStats(rdvs) {
    const enCours = rdvs.filter(r => r.statut === 'en_cours').length;
    const termines = rdvs.filter(r => r.statut === 'termine' || r.statut === 'facture').length;
    
    document.getElementById('stat-en-cours').textContent = enCours;
    document.getElementById('stat-termines').textContent = termines;
}

// Sélectionner un RDV
async function selectRDV(rdvId) {
    currentRdvId = rdvId;
    
    // Mettre à jour l'UI
    document.querySelectorAll('.rdv-card').forEach(card => {
        card.classList.remove('active');
        if (parseInt(card.dataset.rdvId) === rdvId) {
            card.classList.add('active');
        }
    });
    
    try {
        // Récupérer les détails du RDV
        const today = new Date().toISOString().split('T')[0];
        const planning = await apiGet(`/api/planning?date=${today}`);
        currentRdv = planning.rendez_vous.find(r => r.id === rdvId);
        
        if (!currentRdv) {
            showError('RDV non trouvé');
            return;
        }
        
        renderRDVDetail(currentRdv);
        
    } catch (error) {
        console.error('Erreur:', error);
        showError('Erreur de chargement du RDV');
    }
}

function renderRDVDetail(rdv) {

    const container = document.getElementById('rdv-detail');
    
    // Cloner le template
    container.innerHTML = '';
    const content = template.content.cloneNode(true);
    
    // Remplir les infos
    content.getElementById('detail-client').textContent = `${rdv.client.prenom} ${rdv.client.nom}`;
    content.getElementById('detail-vehicule').textContent = `${rdv.vehicule.marque} ${rdv.vehicule.modele}`;
    content.getElementById('detail-plaque').textContent = `Plaque: ${rdv.vehicule.plaque}`;
    content.getElementById('detail-heure').textContent = rdv.heure_rdv;
    content.getElementById('detail-duree').textContent = `${rdv.temps_estime || 60} min`;
    
    // Statut
    const statutEl = content.getElementById('detail-statut');
    statutEl.textContent = getStatusLabel(rdv.statut);
    statutEl.className = `px-3 py-1 rounded-full text-sm font-semibold ${getStatusBadgeClass(rdv.statut)}`;
    
    // OR Info
    content.getElementById('detail-km').textContent = rdv.kilometrage || '-';
    content.getElementById('detail-etat').textContent = rdv.etat_vehicule ? 
        (rdv.etat_vehicule.length > 50 ? rdv.etat_vehicule.substring(0, 50) + '...' : rdv.etat_vehicule) : 
        'Non renseigné';
    
    // Initialiser les points de contrôle
    pointsControle = {};
    CHECKLIST_DEFAULT.forEach(item => {
        pointsControle[item.id] = 'non_verifie';
    });
    
    // Générer la checklist avec 3 états
    const checklistContainer = content.getElementById('checklist-container');
    checklistContainer.innerHTML = CHECKLIST_DEFAULT.map(item => `
        <div class="check-item p-3 border rounded-lg">
            <div class="font-medium mb-2">${item.label}</div>

                    Non vérifié
                </button>
                <button type="button" onclick="setCheckState('${item.id}', 'ok')" 
                        id="btn-${item.id}-ok"
                        class="flex-1 px-3 py-1 text-sm rounded border border-gray-300 bg-white text-gray-600 check-btn-${item.id}">
                    ✓ OK
                </button>
                <button type="button" onclick="setCheckState('${item.id}', 'nok')" 
                        id="btn-${item.id}-nok"
                        class="flex-1 px-3 py-1 text-sm rounded border border-gray-300 bg-white text-gray-600 check-btn-${item.id}">
                    ✗ Pas OK
                </button>
            </div>
        </div>

    loadExistingRapport(rdv.id);
    
    // Charger les infos de temps de travail

    piecesUtilisees = [];
    renderPiecesList();

// Charger un rapport existant
async function loadExistingRapport(rdvId) {
    try {
        const response = await fetch(`${API_URL}/api/rendez-vous/${rdvId}/rapport-technicien`, {
            headers: { 'Authorization': `Bearer ${getToken()}` }
        });
        
        if (response.ok) {
            const rapport = await response.json();
            
            // Restaurer les points de contrôle
            if (rapport.points_controle) {
                Object.entries(rapport.points_controle).forEach(([id, state]) => {
                    if (pointsControle.hasOwnProperty(id)) {
                        setCheckState(id, state, true);
                    }
                });
            }
            
            // Restaurer les autres champs
            const alertesEl = document.getElementById('tech-alertes');
            const recommandationsEl = document.getElementById('tech-recommandations');
            const travauxEl = document.getElementById('tech-travaux');
            
            if (alertesEl && rapport.alertes) alertesEl.value = rapport.alertes;
            if (recommandationsEl && rapport.recommandations) recommandationsEl.value = rapport.recommandations;
            if (travauxEl && rapport.travaux_realises) travauxEl.value = rapport.travaux_realises;
            
            // Restaurer les pièces
            if (rapport.pieces_utilisees && rapport.pieces_utilisees.length > 0) {
                piecesUtilisees = rapport.pieces_utilisees;
                renderPiecesList();
            }
        }
    } catch (error) {
        console.log('Pas de rapport existant');
    }
}

// Définir l'état d'un point de contrôle
function setCheckState(itemId, state, updateVisual = true) {
    pointsControle[itemId] = state;
    
    if (updateVisual) {
        // Mettre à jour l'apparence des boutons
        const states = ['non_verifie', 'ok', 'nok'];
        const colors = {
            'non_verifie': 'bg-gray-100 text-gray-600 border-gray-300',
            'ok': 'bg-green-100 text-green-700 border-green-400 font-semibold',
            'nok': 'bg-red-100 text-red-700 border-red-400 font-semibold'
        };
        
        states.forEach(s => {
            const btn = document.getElementById(`btn-${itemId}-${s}`);
            if (btn) {
                // Reset classes
                btn.className = `flex-1 px-3 py-1 text-sm rounded border check-btn-${itemId} ${colors[s === state ? state : 'non_verifie']}`;
                if (s !== 'non_verifie' && s !== state) {
                    btn.className = `flex-1 px-3 py-1 text-sm rounded border border-gray-300 bg-white text-gray-600 check-btn-${itemId}`;
                }
            }
        });
    }
}

function toggleAllChecks() {
    const checkboxes = document.querySelectorAll('#checklist-container input[type="checkbox"]');
    const allChecked = Array.from(checkboxes).every(cb => cb.checked);
    checkboxes.forEach(cb => cb.checked = !allChecked);
}

// Gestion des pièces
function addPiece() {
    const nomInput = document.getElementById('piece-nom');
    const qtyInput = document.getElementById('piece-qty');
    
    const nom = nomInput.value.trim();
    const qty = parseInt(qtyInput.value) || 1;
    
    if (!nom) {
        alert('Veuillez saisir le nom de la pièce');
        return;
    }
    
    piecesUtilisees.push({ nom, qty });
    renderPiecesList();
    
    // Reset inputs
    nomInput.value = '';
    qtyInput.value = '1';
    nomInput.focus();
}

function removePiece(index) {
    piecesUtilisees.splice(index, 1);
    renderPiecesList();
}

function renderPiecesList() {
    const container = document.getElementById('pieces-list');
    if (!container) return;
    
    if (piecesUtilisees.length === 0) {
        container.innerHTML = '<p class="text-gray-500 text-sm italic">Aucune pièce ajoutée</p>';
        return;
    }
    
    container.innerHTML = piecesUtilisees.map((piece, index) => `
        <div class="flex justify-between items-center p-2 bg-gray-50 rounded">
            <span>${piece.qty}x ${piece.nom}</span>
            <button onclick="removePiece(${index})" class="text-red-500 hover:text-red-700">
                <i class="fas fa-trash"></i>
            </button>
        </div>
    `).join('');
}

// Sauvegarder le rapport technique
async function saveTechReport() {
    if (!currentRdvId) return;

    // Récupérer les données
    const alertes = document.getElementById('tech-alertes')?.value || '';
    const recommandations = document.getElementById('tech-recommandations')?.value || '';
    const travaux = document.getElementById('tech-travaux')?.value || '';

    // Préparer les pièces avec référence si disponible
    const piecesAvecRef = piecesUtilisees.map(p => ({
        nom: p.nom,
        quantite: p.qty,
        reference: p.reference || ''
    }));

    const rapportData = {
        points_controle: pointsControle,
        alertes: alertes,
        recommandations: recommandations,
        travaux_realises: travaux,
        pieces_utilisees: piecesAvecRef,
        statut: 'en_cours'
    };

    try {
        const response = await fetch(`${API_URL}/api/rendez-vous/${currentRdvId}/rapport-technicien`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${getToken()}`
            },
            body: JSON.stringify(rapportData)
        });

        if (!response.ok) throw new Error('Erreur lors de la sauvegarde');

        // Mettre à jour le statut du RDV si nécessaire
        if (currentRdv.statut === 'en_attente') {
            await apiPut(`/api/rendez-vous/${currentRdvId}`, { statut: 'en_cours' });
        }

        showSuccess('Rapport sauvegardé !');
        loadTechRDV(); // Rafraîchir la liste

    } catch (error) {
        console.error('Erreur sauvegarde:', error);
        showError('Erreur lors de la sauvegarde');
    }
}

// Marquer comme terminé ou en attente de pièce
async function markAsDone(attentePiece = false) {
    if (!currentRdvId) return;

    const message = attentePiece
        ? 'Confirmer la mise en attente pour pièce manquante ?'
        : 'Confirmer la fin des travaux ?';

    if (!confirm(message)) return;

    // D'abord sauvegarder le rapport
    const alertes = document.getElementById('tech-alertes')?.value || '';
    const recommandations = document.getElementById('tech-recommandations')?.value || '';
    const travaux = document.getElementById('tech-travaux')?.value || '';

    const piecesAvecRef = piecesUtilisees.map(p => ({
        nom: p.nom,
        quantite: p.qty,
        reference: p.reference || ''
    }));

    const rapportData = {
        points_controle: pointsControle,
        alertes: alertes,
        recommandations: recommandations,
        travaux_realises: travaux,
        pieces_utilisees: piecesAvecRef,
        statut: attentePiece ? 'attente_piece' : 'termine'
    };

    try {
        // Sauvegarder le rapport
        await fetch(`${API_URL}/api/rendez-vous/${currentRdvId}/rapport-technicien`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${getToken()}`
            },
            body: JSON.stringify(rapportData)
        });

        // Mettre à jour le statut du RDV
        await apiPut(`/api/rendez-vous/${currentRdvId}`, {
            statut: attentePiece ? 'en_cours' : 'termine'
        });

        showSuccess(attentePiece ? 'RDV en attente de pièce !' : 'RDV marqué comme terminé !');
        loadTechRDV();

    } catch (error) {
        console.error('Erreur:', error);
        showError('Erreur lors de la mise à jour');
    }
}

// ========== SUIVI TEMPS DE TRAVAIL ==========
let tempsTravailInterval = null;

// Charger les infos de temps de travail
async function loadTempsTravail() {
    if (!currentRdvId) return;
    
    try {
        const response = await fetch(`${API_URL}/api/rendez-vous/${currentRdvId}/temps-travail`, {
            headers: { 'Authorization': `Bearer ${getToken()}` }
        });
        
        if (response.ok) {
            const data = await response.json();
            updateTempsTravailUI(data);
        }
    } catch (error) {
        console.log('Pas d\'info temps de travail');
    }
}

// Mettre à jour l'affichage du temps de travail
function updateTempsTravailUI(data) {
    const section = document.getElementById('temps-travail-section');
    const btnDemarrer = document.getElementById('btn-demarrer');
    const btnTerminer = document.getElementById('btn-terminer');
    const display = document.getElementById('temps-travail-display');
    const info = document.getElementById('temps-travail-info');
    
    if (!section) return;
    section.classList.remove('hidden');
    
    // Arrêter le timer existant
    if (tempsTravailInterval) {
        clearInterval(tempsTravailInterval);
        tempsTravailInterval = null;
    }
    
    if (data.en_cours) {
        // Travail en cours
        btnDemarrer.classList.add('hidden');
        btnTerminer.classList.remove('hidden');
        
        // Parser la date ISO correctement
        const debut = new Date(data.heure_debut_travail + 'Z'); // Force UTC
        info.textContent = 'Démarré à ' + debut.toLocaleTimeString('fr-FR');
        info.classList.remove('hidden');
        
        // Mettre à jour le chrono en temps réel
        tempsTravailInterval = setInterval(() => {
            const now = new Date();
            const diffMs = now.getTime() - debut.getTime();
            const diff = Math.floor(diffMs / 1000 / 60); // minutes
            const heures = Math.floor(diff / 60);
            const mins = diff % 60;
            display.textContent = `${heures}h ${mins.toString().padStart(2, '0')}m`;
        }, 1000);
        
    } else if (data.temps_effectif_minutes) {
        // Travail terminé
        btnDemarrer.classList.add('hidden');
        btnTerminer.classList.add('hidden');
        
        const heures = Math.floor(data.temps_effectif_minutes / 60);
        const mins = data.temps_effectif_minutes % 60;
        display.textContent = `${heures}h ${mins.toString().padStart(2, '0')}m`;
        
        const debut = new Date(data.heure_debut_travail + 'Z');
        const fin = new Date(data.heure_fin_travail + 'Z');
        info.textContent = `De ${debut.toLocaleTimeString('fr-FR')} à ${fin.toLocaleTimeString('fr-FR')}`;
        info.classList.remove('hidden');
        
    } else {
        // Pas encore démarré
        btnDemarrer.classList.remove('hidden');
        btnTerminer.classList.add('hidden');
        display.textContent = '--:--';
        info.classList.add('hidden');
    }
}

// Démarrer le travail
async function demarrerTravail() {
    if (!currentRdvId) return;
    
    try {
        const response = await fetch(`${API_URL}/api/rendez-vous/${currentRdvId}/demarrer-travail`, {
            method: 'POST',
            headers: { 'Authorization': `Bearer ${getToken()}` }
        });
        
        if (response.ok) {

            showSuccess('Travail démarré !');
            loadTempsTravail();

            throw new Error('Erreur lors du démarrage');

    } catch (error) {
        showError('Impossible de démarrer le travail');
    }
}

// Terminer le travail
async function terminerTravail() {
    if (!currentRdvId) return;
    
    try {
        const response = await fetch(`${API_URL}/api/rendez-vous/${currentRdvId}/terminer-travail`, {
            method: 'POST',
            headers: { 'Authorization': `Bearer ${getToken()}` }
        });
        
        if (response.ok) {
            const data = await response.json();
            showSuccess(`Travail terminé ! Temps: ${Math.floor(data.temps_effectif_minutes / 60)}h ${data.temps_effectif_minutes % 60}m`);
            loadTempsTravail();
        } else {
            throw new Error('Erreur lors de la terminaison');
        }
    } catch (error) {
        showError('Impossible de terminer le travail');
    }
}

// Utilitaires
function showSuccess(message) {
    // Simple alert pour l'instant, peut être remplacé par un toast
    alert(message);
}

function showError(message) {
    alert('Erreur: ' + message);
}
